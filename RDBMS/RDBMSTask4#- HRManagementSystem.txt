   SELECT d.DepartmentName, SUM(p.Amount) AS TotalPayroll
    -> FROM Payroll p
    -> JOIN Employees e ON p.EmployeeID = e.EmployeeID
    -> JOIN Departments d ON e.DepartmentID = d.DepartmentID
    -> GROUP BY d.DepartmentName;
